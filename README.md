## Install

```shell
npm install find-hanzi
```

## Usage

```js
const findHanzi = require('find-hanzi')

findHanzi('我')
.then(console.log)
.catch(console.error)
// [ {
//   hanzi: '我',
//   pinyin: 'wǒ',
//   pinyinList: ['wǒ'],
//   cangjie: '竹手戈',
//   cangjieRomanized: 'HQI',
//   strokes: '7',
//   frequency: '1',
//   definition: 'our, us, i, me, my, we'
// } ]

findHanzi('AMYO')
.then(console.log)
.catch(console.error)
// [ {
//   hanzi: '是',
//   pinyin: 'shì, tí',
//   pinyinList: ['shì', 'tí'],
//   cangjie: '日一卜人',
//   cangjieRomanized: 'AMYO',
//   strokes: '9',
//   frequency: '1',
//   definition: 'indeed, yes, right; to be; demonstrative pronoun, this, that'
// } ]
```
